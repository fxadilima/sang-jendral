
export function ToggleButton(props) {

  );
}

export function CardHeader(props) {
  return <div class="card-header"><h1>{props.title}</h1>{props.children}</div>;
}

# Membuat Card dan Panel

Ini Card yang bisa kita buat untuk panel sebelah:

```html
<div class="card">
  <div class="card-header">
    <div class="btn-group">
        <button class="btn btn-primary" onClick={
            async () => {
                const data = await fetch("/components/sidebar.md");
                const text = await data.text();
                document.getElementById('testTag').innerHTML = text;
            }}>Test Button</button>

        <button class="btn btn-primary" onClick={
            () => { document.getElementById('testTag').innerHTML = "" }
        }>Clear</button>
    </div>
  </div>
  <div class="card-body">
    <pre id="testTag"></pre>
  </div>
</div>
```
